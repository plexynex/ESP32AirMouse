        error = Wire.endTransmission();

        if (error == 0) {
            Serial.print("âœ… Perangkat ditemukan di alamat 0x");
            Serial.println(address, HEX);
            nDevices++;
        } else if (error == 4) {
            Serial.print("âš ï¸ Kesalahan di alamat 0x");
            Serial.println(address, HEX);
        }
    }
